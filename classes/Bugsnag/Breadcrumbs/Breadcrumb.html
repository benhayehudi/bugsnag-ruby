def initialize(name, type, meta_data, auto)
  @should_ignore = false
  self.name = name
  self.type = type
  self.meta_data = meta_data

  # Use the symbol comparison to improve readability of breadcrumb creation
  @auto = auto == :auto

  # Store it as a timestamp for now
  @timestamp = Time.now.utc
end
